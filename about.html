function toggleLanguage() { const elements = document.querySelectorAll("[data-th]"); const button = document.querySelector(".lang-btn"); elements.forEach(el => { const isHTML = el.dataset.th.includes("<"); // เช็คว่ามี HTML ไหม if (currentLang === "th") { if (el.dataset.en) { if (isHTML) { el.innerHTML = el.dataset.en; } else { el.textContent = el.dataset.en; } } } else { if (el.dataset.th) { if (isHTML) { el.innerHTML = el.dataset.th; } else { el.textContent = el.dataset.th; } } } }); if (currentLang === "th") { document.documentElement.lang = "en"; button.innerText = "TH"; currentLang = "en"; } else { document.documentElement.lang = "th"; button.innerText = "EN"; currentLang = "th"; } } } }
